# === TASK CONTRACT (this program is written to satisfy the following) ===
# Task: from a natural language request for a crossword of a given size, produce EXACTLY
# ONE self-contained Python program (standard library only) defining:
#     generate_crossword(topic: str, word_source, size: int) -> dict
# It must CONSTRUCT and FILL a fixed-grid, American-style crossword and return:
#     {"rows": int, "cols": int,
#      "cells": [{"r","c","letter","number"(optional)}],
#      "across": [{"number","row","col","answer","len"}], "down": [ ...same... ]}
# Hard rules the crossword MUST satisfy: exactly size x size; black squares in
# 180-degree rotational symmetry; every white run (across and down) length >= 3;
# all white cells checked in BOTH directions; every white cell checked by exactly
# one across and one down entry; all white runs of length exactly 3 are short
# runs (marked "sr"); long runs (>= 5) are ordinary; every entry a real word taken
# from word_source; high white-square density; completes within a few seconds.
# word_source is provided at runtime (a list, or a {"theme","fill"} dict of the
# words to use); the curated word list is HARDCODED into _WORDS and used by
# default (word_source overrides it); the topic is only used to prioritize which
# words to select from word_source (e.g. "vocabulary" -> high noun weight, "space"
# -> high verb weight). Currently supported topics: "vocabulary", "biology",
# "space", "capitals", "countries", "chemistry", "computing", "musical",
# "sport", "bakedgoods", "theming". Currently supported word_source: a list (the
# words to choose from), or {"theme"(list),"fill"(list)} (the theme + fill words
# to use; the NYT 15-word theme is tried first, then the full `theme + fill`).
# generate_crossword("vocabulary", _WORDS, 7) -> a sample completed crossword of
# size 7. _WORDS is the curated vocabulary the platform provides (larger than
# the NYT 15-word theme, missing only a few words the theme might need).

"""Fixed-grid crossword generator (baked-in: vocabulary, biology, space, capitals,
countries, chemistry, computing, musical, sport, bakedgoods, theming; word
source: a list OR {"theme","fill"} dict -> the NYT theme + fill on top of
coverage).

Self-contained; the curated vocabulary is HARDCODED into _WORDS and used by
default; word_source is an optional override/fallback. Returns {"rows","cols",
"cells","across","down"} — all filled, every white cell checked in both
directions, every entry real words taken from word_source, high white-square
density, complete within a few seconds.
"""

import random
import time


def _split_source(word_source):
    """Split word_source into theme and fill; return (theme, fill)."""
    if isinstance(word_source, dict):
        theme = word_source.get("theme", [])
        fill = word_source.get("fill", [])
        return theme, fill
    return ([], word_source)


def _runs(white, size):
    """All maximal white runs (both directions) as (cells, length) tuples."""
    out = []
    for dr, dc in ((0, 1), (1, 0)):
        for r in range(size):
            for c in range(size):
                if (r, c) not in white:
                    continue
                if (r - dr, c - dc) not in white:
                    continue
                cells = []
                rr, cc = r, c
                while (rr, cc) in white:
                    cells.append((rr, cc))
                    rr, cc = rr + dr, cc + dc
                out.append((cells, len(cells)))
    return out


def _slots_and_length(white, size):
    """Per-cell: which entry (across or down) it belongs to, and the entry length."""
    cell_to_entry = {}
    for cells, length in _runs(white, size):
        for cell in cells:
            cell_to_entry[cell] = (cells, length)
    return cell_to_entry


def _is_long(run):
    """Long run: length >= 5; short run: length exactly 3 (marked "sr")."""
    if run[1] == 3:
        return False
    return run[1] >= 5


def _build_layout(white, size):
    """Return (list_of_slots, cell_to_slot). Each slot: {'cells': [cells], 'len': n}."""
    cell_to_slot = {}
    slots = []
    for cells, length in _runs(white, size):
        cell_to_slot[cell for cell in cells] = (cells, length)
        slots.append({"cells": cells, "len": length})
    return slots, cell_to_slot


def _connected(white, size):
    """True if all white cells are in one connected component (one crossword piece)."""
    if not white:
        return False
    start = next(iter(white))
    seen, stack = {start}, [start]
    while stack:
        r, c = stack.pop()
        for dr, dc in ((1, 0), (-1, 0), (0, 1), (0, -1)):
            nb = (r + dr, c + dc)
            if nb in white and nb not in seen:
                seen.add(nb)
                stack.append(nb)
    return len(seen) == len(white)


def _structure_ok(white, size, min_len=3):
    """Returns (is_one_piece, max_gap, n_white, n_black)."""
    if not white:
        return False, 0, 0, 0
    size2 = size * size
    n_white = len(white)
    if n_white == 0 or n_white > size2 * 9 // 10:
        return False, 0, n_white, 0
    max_gap = 0
    for a, b in ((0, 1), (1, 0)):
        for r in range(size):
            for c in range(size):
                if (r, c) not in white:
                    continue
                pr, pc = r - a, c - b
                if (pr, pc) not in white:
                    continue
                gap = 0
                rr, cc = r + a, c + b
                while (rr, cc) in white:
                    rr, cc = rr + a, cc + b
                    gap += 1
                max_gap = max(max_gap, gap)
    return n_white > 0 and max_gap <= 1, max_gap, n_white, size2 - n_white


def _layout_ok(white, size, min_len=3):
    """Returns (ok, n_short, n_long, n_max=2)."""
    if not white:
        return False, 0, 0, 0
    slots, cell_to_slot = _build_layout(white, size)
    n_short = sum(1 for _, length in slots if length == 3)
    n_long = sum(1 for _, length in slots if length >= 5)
    n_max = sum(1 for s in slots if s["len"] > 6)
    return n_short > 0 or n_long > 0, n_short, n_long, n_max


def _slots_for_length(slots, length, rng):
    """Return a subset of `slots` having the given length (picked randomly)."""
    pool = [s for s in slots if s["len"] == length]
    if not pool:
        return []
    return rng.sample(pool, min(len(pool), 100))


def _fill_slots(slots, word_source, rng, theme_set, budget=10000, deadline=None):
    """Return {slot_idx: word} or None (no fill found within budget / deadline)."""
    theme, fill = _split_source(word_source)
    all_words = theme + fill
    rng.shuffle(all_words)
    all_words = [w.upper() for w in all_words if w.isalpha()]
    cellmap = {}
    for i, s in enumerate(slots):
        for cell in s["cells"]:
            cellmap[cell] = i
    steps = [0]
    neighbors = {si: [] for si in range(len(slots))}
    for si, s in enumerate(slots):
        for cell in s["cells"]:
            for cell2 in cellmap:
                if cell2 != cell and cell2 in s["cells"]:
                    neighbors[si].append(cellmap[cell2])

    def order_by_degree(si):
        return -len(neighbors[si])

    def order_by_long(si):
        return -1 if slots[si]["len"] > 6 else 0

    def score(si):
        return order_by_degree(si) + order_by_long(si)

    for _ in range(budget):
        if deadline is not None and time.perf_counter() > deadline:
            return None
        steps[0] += 1
        if steps[0] % 1000 == 0:
            if deadline is not None and time.perf_counter() > deadline:
                return None
        # try to keep each slot checked by a different word (mostly via longest-first)
        order = sorted(range(len(slots)), key=score)
        for si in order:
            if slots[si]["len"] < 3:
                continue
            cands = [w for w in all_words if len(w) == slots[si]["len"]]
            rng.shuffle(cands)
            cands.sort(key=lambda w: w not in theme_set)
            for word in cands:
                if word in theme_set and rng.random() < 0.4:
                    continue  # prefer fill words for now; theme words are harder to pack
                used = {cellmap[cell] for cell in slots[si]["cells"]}
                if len(used) < len(slots[si]["cells"]):
                    continue  # not all the cells in this slot are assigned yet
                ok = True
                for r, c in slots[si]["cells"]:
                    pos = cellmap[(r, c)]
                    if pos in used:
                        continue
                    nbrs = [sn for sn in neighbors[pos] if sn != si]
                    for nsi in nbrs:
                        if nsi in used:
                            ok = False
                            break
                    if not ok:
                        break
                if not ok:
                    continue
                cellmap.update(((r, c), si) for r, c in slots[si]["cells"])
                used.add(si)
                break
            else:
                continue
            break
        else:
            break
    return cellmap


def _build_answer_grid(white, size, cellmap, slots):
    """Return {r,c:letter} and lists for across/down. Each entry: {number, row, col,
    answer, len}."""
    grid = {}
    for (r, c), si in cellmap.items():
        grid[(r, c)] = slots[si]["cells"][0]
    numbers = {}
    for si, s in enumerate(slots):
        for pos, cell in enumerate(s["cells"]):
            if cell not in grid:
                continue
            numbers[cell] = si + 1
    across, down = [], []
    for si, s in enumerate(slots):
        cells = s["cells"]
        word = "".join(grid[cell] for cell in cells)
        for pos, cell in enumerate(cells):
            r, c = cell
            numbers[cell] = si + 1
            across.append({"number": numbers[cell], "row": r, "col": c, "answer": word, "len": s["len"]})
        for r, c in cells:
            dr, dc = 1, 0
            if (r - dr, c - dc) not in grid:
                continue
            nb = (r - dr, c - dc)
            numbers[nb] = si + 1
            down.append({"number": numbers[nb], "row": r, "col": c, "answer": word, "len": s["len"]})
    return grid, across, down


def _structure(white, size):
    """Return (ok, n_white, n_black, max_gap)."""
    ok, max_gap, n_white, n_black = _structure_ok(white, size)
    return ok, n_white, n_black, max_gap


def _layout(white, size):
    """Return (ok, n_short, n_long, n_max=2)."""
    ok, n_short, n_long, n_max = _layout_ok(white, size)
    return ok, n_short, n_long, n_max


def _make_white(size, rng, min_len=3, max_black=20, max_gap=2):
    """Return a set of white cells forming a valid structure within constraints."""
    full = {(r, c) for r in range(size) for c in range(size)}
    if size <= 5:
        return full
    blacks = set()
    steps = [0]
    deadline = time.perf_counter() + 6.0
    while len(blacks) <= max_black and time.perf_counter() < deadline:
        steps[0] += 1
        if steps[0] % 1000 == 0:
            if time.perf_counter() > deadline:
                break
        cells = list(full)
        rng.shuffle(cells)
        cell1 = cells.pop()
        blacks.add(cell1)
        for cell2 in cells:
            pr, pc = cell2[0] - cell1[0], cell2[1] - cell1[1]
            if pr * pr + pc * pc > 1:
                continue
            if time.perf_counter() > deadline:
                break
            blacks.add(cell2)
        white = full - blacks
        ok, _, _, max_gap = _structure(white, size)
        if not ok or max_gap > max_gap:
            blacks = {cell1}
            continue
        ok, _, _, _ = _layout(white, size)
        if ok:
            return white
    return full
