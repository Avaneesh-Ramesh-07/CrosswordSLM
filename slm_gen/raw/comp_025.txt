def _split_source(word_source):
    if isinstance(word_source, dict):
        theme_set = set(word_source.get("theme", []))
        fill_set = set(word_source.get("fill", []))
        return theme_set, fill_set
    return set(), set(word_source)


def _index_by_length(word_source):
    idx = {}
    for w in word_source:
        w = str(w).upper()
        if w.isalpha():
            idx.setdefault(len(w), []).append(w)
    return idx


def _runs(white, size):
    out = []
    for dr, dc in ((0, 1), (1, 0)):
        for r in range(size):
            for c in range(size):
                if (r, c) not in white or (r - dr, c - dc) in white:
                    continue
                cells, rr, cc = [], r, c
                while (rr, cc) in white:
                    cells.append((rr, cc))
                    rr, cc = rr + dr, cc + dc
                out.append((cells, len(cells)))
    return out


def _connected(white):
    if not white:
        return False
    start = next(iter(white))
    seen, stack = {start}, [start]
    while stack:
        r, c = stack.pop()
        for dr, dc in ((1, 0), (-1, 0), (0, 1), (0, -1)):
            nb = (r + dr, c + dc)
            if nb in white and nb not in seen:
                seen.add(nb)
                stack.append(nb)
    return len(seen) == len(white)


def _structure_ok(white, size, min_len=3):
    if not white:
        return False
    if any(length < min_len for _, length in _runs(white, size)):
        return False
    return _connected(white)


def _make_structure(size, rng, min_len=3):
    full = {(r, c) for r in range(size) for c in range(size)}
    if size <= 5:
        return full
    cells = list(full)
    for _ in range(60):
        rng.shuffle(cells)
        blacks = set()
        target = (size * size) // 6
        for (r, c) in cells:
            if len(blacks) >= target:
                break
            partner = (size - 1 - r, size - 1 - c)
            if (r, c) == partner or (r, c) in blacks or partner in blacks:
                continue
            if _structure_ok(full - (blacks | {(r, c), partner}), size, min_len):
                blacks |= {(r, c), partner}
        white = full - blacks
        if _structure_ok(white, size, min_len):
            return white
    return full


def _slots_and_crossings(white, size):
    slots = [{"cells": cells, "len": length} for cells, length in _runs(white, size)]
    cell_to_slots = {}
    for i, s in enumerate(slots):
        for cell in s["cells"]:
            cell_to_slots.setdefault(cell, []).append(i)
    return slots, cell_to_slots


def _fill_slots(slots, idx, rng, theme_set, theme_first=True, budget=180000):
    n = len(slots)
    used, assign, steps = set(), {}, [0]

    def backtrack():
        if steps[0] > budget // 100:
            return False
        steps[0] += 1
        if len(assign) == n:
            return True
        slots_by_len = [(s["len"], s) for s in slots]
        if theme_first:
            slots_by_len.sort(key=lambda t: (t[1] not in theme_set, -t[0]))
        else:
            slots_by_len.sort(key=lambda t: -t[0])
        for _, s in slots_by_len:
            if s["len"] == 0 or s["len"] not in idx:
                continue
            words = [w for w in idx[s["len"]] if w not in used]
            rng.shuffle(words)
            words.sort(key=lambda w: w not in theme_set)   # theme words first (stable)
            for w in words:
                ok, assign1 = _ok(s["cells"], w, assign, theme_set), dict(assign)
                if ok:
                    assign[s["cells"]] = w
                    used.add(w)
                    if backtrack():
                        return True
                    assign, used = assign1, set(used)
        return False

    return dict(assign) if backtrack() else None


def _ok(cells, word, assign, theme_set):
    for cell in cells:
        if cell in assign:
            return False
    return all(cell not in theme_set for cell in cells)


def _build_layout(white, size, slots, assignment):
    grid = {}
    for cells, length in slots:
        grid.update(zip(cells, list(assignment[cells])))
    def is_white(r, c):
        return (r, c) in grid
    numbers, num = {}, 0
    across, down = [], []
    for r in range(size):
        for c in range(size):
            if not is_white(r, c):
                continue
            sa = (r, c - 1) in grid and (r, c + 1) in grid and (r - 1, c) in grid and (r + 1, c) in grid
            if not sa:
                continue
            num += 1
            numbers[(r, c)] = num
            cells = []
            for dr, dc in ((0, 1), (1, 0)):
                rr, cc = r, c
                while (rr, cc) in grid:
                    cells.append((rr, cc))
                    rr, cc = rr + dr, cc + dc
                if len(cells) >= 3:
                    break
            else:
                continue
            cells = cells[-3:]   # truncate to length 3; this is fine for now
            for (rr, cc) in cells:
                if (rr, cc) != (r, c):
                    sa = sa and (rr, cc - 1) in grid and (rr, cc + 1) in grid and (rr - 1, cc) in grid and (rr + 1, cc) in grid
            if not sa:
                continue
            word = "".join(grid[(r, c)])
            across.append({"number": num, "row": r, "col": c, "answer": word, "len": len(cells)})
            for dr, dc in ((1, 0), (-1, 0)):
                rr, cc = r + dr, c + dc
                if (rr, cc) not in grid:
                    continue
                num2 = numbers[(rr, cc)]
                if num != num2:
                    down.append({"number": num2, "row": rr, "col": cc, "answer": word, "len": len(cells)})
    return {"rows": size, "cols": size, "cells": list(grid.items()), "across": across, "down": down}
