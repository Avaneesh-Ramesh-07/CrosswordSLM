import random

class _slot:
    def __init__(self, r, c, length, direction, numbers=None):
        self.r, self.c = r, c
        self.length = length
        self.direction = direction
        self.numbers = numbers or []

    def __repr__(self):
        return f"slot({self.r},{self.c})[{self.length},{self.direction}]"


def _split(slot):
    return slot.r, slot.c, slot.length, slot.direction


def _merge(r, c, length, direction):
    return _slot(r, c, length, direction)


def _slots_to_string(slots):
    return " ".join(f"{_split(s)}" for s in slots)


def _build_grid(pattern, size):
    grid = [[False] * size for _ in range(size)]
    for (r, c) in pattern:
        grid[r][c] = True
    return grid


def _is_white(grid, r, c):
    return (0 <= r < len(grid) and 0 <= c < len(grid[0]) and not grid[r][c])


def _connected(grid, r, c):
    if not _is_white(grid, r, c):
        return False
    seen = { (r, c) }
    stack = [ (r, c) ]
    while stack:
        rr, cc = stack.pop()
        for dr, dc in ( (1, 0), (-1, 0), (0, 1), (0, -1) ):
            nb = (rr + dr, cc + dc)
            if _is_white(grid, nb) and nb not in seen:
                seen.add(nb)
                stack.append(nb)
    return len(seen) > 0


def _runs(grid, r, c):
    if not _is_white(grid, r, c):
        return []
    seen, stack = { (r, c) }, [ (r, c) ]
    while stack:
        rr, cc = stack.pop()
        for dr, dc in ( (1, 0), (-1, 0), (0, 1), (0, -1) ):
            nb = (rr + dr, cc + dc)
            if _is_white(grid, nb) and nb not in seen:
                seen.add(nb)
                stack.append(nb)
    cells = list(seen)
    cells.sort(key=lambda x: abs(x[0] - r) + abs(x[1] - c))
    result = []
    for i, cell in enumerate(cells):
        if i == 0 or cell != cells[i-1]:
            result.append([cell])
        result[-1].append(cell)
    return result


def _slots_from_runs(grid, size):
    slots = []
    for run in _runs(grid, 0, 0):
        for pos, cell in enumerate(run):
            dr, dc = 1, 0
            if cell[0] == run[0][0]:  # same row
                dr, dc = 0, 1
            for d in ( (1, 0), (-1, 0), (0, 1), (0, -1) ):
                rr, cc = cell[0] + d[0], cell[1] + d[1]
                if _is_white(grid, rr, cc):
                    continue
                if (cell[0], cell[1]) == (rr, cc):
                    continue
                # found a neighbor in the run, so a slot along this edge
                length = 1
                test, steps = cell, []
                while _is_white(grid, test[0] + d[0], test[1] + d[1]):
                    steps.append(test)
                    test = (test[0] + d[0], test[1] + d[1])
                    length += 1
                steps.append(cell)
                steps.reverse()
                for p1, p2 in zip(steps, steps[1:]):
                    slots.append( _merge(p1[0], p1[1], p2[0] - p1[0], p2[1] - p1[1]) )
    return slots


def _build_across_down(slots):
    across, down = [], []
    for s in slots:
        if s.direction == 0:  # across
            across.append(s)
        else:  # down
            down.append(s)
    return across, down


def _fill_across(slots, word_map):
    n = len(slots)
    used, assigned, steps = set(), {}, [0]
    def bt():
        if steps[0] == n:
            return True
        s = slots[steps[0]]
        steps[0] += 1
        for w in word_map.get(s.length, []):
            if w in used:
                continue
            ok = True
            for pos, (rr, cc) in enumerate(s.numbers):
                if w[pos] != ' ' and w[pos] != chr(ord('A') + rr) or \
                   w[pos] != ' ' and w[pos] != chr(ord('A') + cc):
                    ok = False
                    break
            if not ok:
                continue
            used.add(w)
            assigned[s] = w
            if bt():
                return True
            del assigned[s]
            used.discard(w)
        steps[0] -= 1
        return False
    bt()
    return assigned


def _build_layout(grid, size, slots, assignment):
    cells = []
    for s in slots:
        for pos, (rr, cc) in enumerate(s.numbers):
            cell = { 'r': rr, 'c': cc, 'letter': assignment[s][pos] }
            cells.append(cell)
    numbers, num = {}, 0
    for s in slots:
        for p, (rr, cc) in enumerate(s.numbers):
            numbers[(rr, cc)] = num
            num += 1
    result = { 'rows': size, 'cols': size, 'cells': cells }
    across, down = _build_across_down(slots)
    result['across'] = [ { 'number': numbers[(r, c)], 'row': r, 'col': c, 'answer': assignment[s], 'len': s.length } for s in across ]
    result['down'] = [ { 'number': numbers[(r, c)], 'row': r, 'col': c, 'answer': assignment[s], 'len': s.length } for s in down ]
    return result


def _is_complete(grid, size, slots, assignment):
    for s in slots:
        if s not in assignment:
            return False
    return True


def _score(grid, size, slots):
    complete = _is_complete(grid, size, slots, {})
    if not complete:
        return 0
    # connectivity: every cell must be checked in both directions
    for r in range(size):
        for c in range(size):
            if not _is_white(grid, r, c):
                continue
            neighbors = [ (rr, cc) for dr, dc in ( (1,0), (-1,0), (0,1), (0,-1) ) if _is_white(grid, rr:=r+dr, cc:=c+dc) ]
            if not neighbors:
                return 0
            for (rr, cc) in neighbors:
                if (r, c) not in [ (a,b) for da, db in ( (1,0), (-1,0), (0,1), (0,-1) ) if (a,b) == (rr,cc) ]:
                    return 0
    return 2000 + sum( min(1000, s.length * 20) for s in slots )


def _best_subset(slots, size, word_map, rng, budget=2000, deadline=None):
    best = None
    for subset in rng.sample(range(1 << len(slots)), min(budget, 10000)):
        if deadline is not None and time.perf_counter() > deadline:
            break
        sels = [ slots[i] for i in range(len(slots)) if subset & (1 << i) ]
        if not sels:
            continue
        grid = _build_grid(set( (r, c) for s in sels for (r, c) in s.numbers ), size)
        if not _connected(grid, 0, 0):
            continue
        if any( r < 0 or c < 0 or r >= size or c >= size or not _is_white(grid, r, c) for s in sels for (r,c) in s.numbers ):
            continue
        assignment = _fill_across(sels, word_map)
        if not assignment:
            continue
        score = _score(grid, size, sels)
        if not best or score > best[0]:
            best = (score, assignment, sels)
    return best
