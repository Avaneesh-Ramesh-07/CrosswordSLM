def _split_source(word_source):
    """Split word_source into theme and fill; return (theme_set, fill_set)"""
    if not word_source:
        return set(), set()
    if isinstance(word_source, dict):
        return word_source.get("theme", set()), word_source.get("fill", set())
    return set(), set(word_source)


def _index_by_length(words):
    """-> {length: [words,...]}  (each word is a str; all same length)"""
    idx = {}
    for w in words:
        idx.setdefault(len(w), []).append(w)
    return idx


def _runs(white, size):
    """-> [ (r,c,len) ] all maximal white runs (both directions)"""
    out = []
    for dr, dc in ((0, 1), (1, 0)):
        for r in range(size):
            for c in range(size):
                if (r, c) not in white:
                    continue
                if (r - dr, c - dc) in white:
                    continue  # not a run start
                cells = []
                rr, cc = r, c
                while (rr, cc) in white:
                    cells.append((rr, cc))
                    rr, cc = rr + dr, cc + dc
                out.append((cells, len(cells)))
    return out


def _connected(white):
    """-> True if all white cells are in one connected component"""
    if not white:
        return False
    start = next(iter(white))
    seen, stack = {start}, [start]
    while stack:
        r, c = stack.pop()
        for dr, dc in ((1, 0), (-1, 0), (0, 1), (0, -1)):
            nb = (r + dr, c + dc)
            if nb in white and nb not in seen:
                seen.add(nb)
                stack.append(nb)
    return len(seen) == len(white)


def _structure_ok(white, size, min_len=3):
    """-> True if the white pattern is valid (connected, all runs >= min_len)"""
    if not white:
        return False
    return _connected(white) and all(length >= min_len for _, length in _runs(white, size))


def _make_structure(size, rng, min_len=3):
    """Return a set of white cells forming a valid structure (or empty on failure)"""
    full = {(r, c) for r in range(size) for c in range(size)}
    if size < 5:
        return full  # a seed: small grids are all fine (word square)
    cells = list(full)
    for _ in range(60):  # a few structure attempts
        rng.shuffle(cells)
        blacks = set()
        for (r, c) in cells:
            if (r, c) in blacks or (r, c) in full:
                continue
            trial_white = full - (blacks | {(r, c)})
            if _structure_ok(trial_white, size, min_len):
                blacks |= {(r, c)}
                full = trial_white
        white = full
        if _structure_ok(white, size, min_len):
            return white
    return set()


def _entry_points(white, size):
    """-> [ (r,c), ... ] the white cells with neighbors in both directions"""
    out = []
    for (r, c) in white:
        if not ((r - 1, c) in white or (r + 1, c) in white or (r, c - 1) in white or (r, c + 1) in white):
            continue
        out.append((r, c))
    return out


def _slots_and_crossings(white, size):
    """-> (slots, cell_to_slots) where each slot: {'r','c','len'} and cell_to_slots: {cell: [slots,...]}"""
    slots = []
    for cells, length in _runs(white, size):
        slot = {"r": cells[0][0], "c": cells[0][1], "len": length}
        slots.append(slot)
    cell_to_slots = {}
    for i, slot in enumerate(slots):
        for cell in _runs(white, size)[i][0]:
            cell_to_slots.setdefault(cell, []).append(i)
    return slots, cell_to_slots

def _build_pattern_index(slots, idx_by_len):
    """-> { (length,pos,letter): {word,...} } fast lookup: words matching a slot's current state"""
    idx = {}
    for length, words in idx_by_len.items():
        for w in words:
            for pos, ch in enumerate(w):
                idx.setdefault((length, pos, ch), set()).add(w)
    return idx


def _word_at(slot, pos, grid):
    """Return the current letter in the slot at position pos, or None"""
    if slot["len"] <= pos:
        return None
    rr, cc = slot["r"], slot["c"]
    return grid.get((rr + pos, cc))


def _set_letter(slot, pos, c, grid):
    """Set grid[(r,c)] = c; return False if it conflicts with an existing fixed letter"""
    rr, cc = slot["r"], slot["c"]
    if (rr + pos, cc) in grid and grid[(rr + pos, cc)] != c:
        return False
    grid[(rr + pos, cc)] = c
    return True


def _reset_letter(slot, pos, grid):
    """Remove the letter from grid; safe to call even if it was never set"""
    rr, cc = slot["r"], slot["c"]
    grid.pop((rr + pos, cc), None)


def _fill_slots(slots, cell_to_slots, idx, rng, theme_set, fill_set, deadline=None):
    """Return {slot_num: word} assignment or None if no assignment within deadline"""
    grid = {}
    used, tried = set(), set()
    steps = [0]
    n = len(slots)

    def domain_size(slot):
        """Words left for this slot (ignores fixed letters from earlier assignments)"""
        if slot["len"] not in idx:
            return 0
        total = len(idx[(slot["len"], 0, " ")])  # words of right length (estimate)
        for pos, (rr, cc) in enumerate(_runs(white, size)[slot["len"]][0]):
            if (rr, cc) in grid:
                total -= sum(1 for w in idx[(slot["len"], pos, grid[(rr, cc)])] if w != grid[(rr, cc)])
        return total

    def backtrack():
        """Return True if any slot is filled; False if dead end (exhausted domain per slot)"""
        if steps[0] > 20000 or (deadline is not None and time.perf_counter() > deadline):
            return False
        steps[0] += 1
        for i, s in enumerate(slots):
            if i in used:
                continue
            if domain_size(s) == 0:
                return False
            cands = [w for w in idx.get((s["len"], 0, " ")) or [] if w not in used]
            rng.shuffle(cands)
            for word in cands:
                ok = True
                for pos, (rr, cc) in enumerate(_runs(white, size)[s["len"]][0]):
                    if (rr, cc) in grid and grid[(rr, cc)] != word[pos]:
                        ok = False
                        break
                if not ok:
                    continue
                grid[s["r"] + pos] = word[pos]
                used.add(word)
                if backtrack():
                    return True
                del grid[s["r"] + pos]
                used.discard(word)
        return False

    if backtrack():
        return dict(used)
    return None


def _build_layout(white, size, slots, assignment):
    """-> {num: {r,c,letter}, ...} for each entry (ignores unassigned slots)"""
    out = []
    for num, word in assignment.items():
        r, c = slots[num]["r"], slots[num]["c"]
        cells = [(r + i, c) for i in range(slots[num]["len"])]
        entry = {"number": num, "row": r, "col": c, "answer": word, "len": slots[num]["len"]}
        out.append(entry)
        for rr, cc in cells:
            out.append({"r": rr, "c": cc, "letter": word[rr - r]})
    return out
