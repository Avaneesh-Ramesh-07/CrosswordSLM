def _split(whole, sz):
    out = []
    for i in range(0, len(whole), sz):
        out.append(whole[i:i+sz])
    return out


def _pool(word_source, n=80):
    """Return a pool of candidate words; prefer long words."""
    if word_source == "preset":
        words = WORDS
    elif word_source == "random":
        words = WORDS
        random.shuffle(words)
    else:
        words = word_source
    return words[:n]


def _runs(white, size):
    """Returns a list of (r,c,len) for all maximal runs (both directions)."""
    out = []
    for dr, dc in ((0, 1), (1, 0)):
        for r in range(size):
            for c in range(size):
                if (r, c) not in white:
                    continue
                if (r - dr, c - dc) not in white:
                    continue
                cells = []
                rr, cc = r, c
                while (rr, cc) in white:
                    cells.append((rr, cc))
                    rr, cc = rr + dr, cc + dc
                out.append((cells, len(cells)))
    return out


def _all_run_ok(white, size, min_len=3):
    """All maximal runs length >= min_len? -> (true, [bad_runs]) or (false, [])"""
    bad = []
    for cells, length in _runs(white, size):
        if length < min_len:
            bad.append((cells, length))
    return len(bad) == 0, bad


def _structure_ok(white, size, min_len=3):
    """Fixed-template: 180-degree rotational symmetry; high white-square density."""
    if size * size - len(white) > size * size // 3:  # >33% black
        return False
    for (r, c) in list(white):
        if (size - 1 - r, size - 1 - c) not in white:
            return False
    return True


def _fill(white, size, rng, pool, suffix=""):
    """Return a filled grid (dict) or None."""
    n = len(white)
    if n == 0:

            if not words:

            rng.shuffle(words)
            for w in words:
                if w not in used:

            else:

            break
        else:

        grid = {}
        for si, word in assignment.items():
            for pos, (r, c) in enumerate(assignment[si]["cells"]):

def _all_ok(grid, size):
    """All white cells checked in both directions; every run checked both ways."""
    for (r, c), ch in grid.items():
        if (r - 1, c) not in grid and (r + 1, c) not in grid and (r, c - 1) not in grid and (r, c + 1) not in grid:
            return False
    return True

def _make_structure(size, rng):
    """Return a set of white cells for a fixed, symmetric structure (size x size)."""
    out = set()
    for r in range(size):
        for c in range(size):
            if (r == 1 and c == 1) or (r == 1 and c == 8) or (r == 8 and c == 1) or (r == 8 and c == 8):
                continue
            out.add((r, c))
    return out


def generate_crossword(topic: str, word_source, size: int) -> dict:
    """Fixed-template generator: returns a NYT-legal grid, or {} on failure."""
    if size < 7:
        return {}
    word_source = word_source or "preset"
    rng = random.Random(hash((topic, size)) & 0xFFFFFFFF)
    white = _make_structure(size, rng)
    if not _structure_ok(white, size):
        return {}
    _, bad = _all_run_ok(white, size)
    if bad:
        return {}
    return _build_layout(white, size)
