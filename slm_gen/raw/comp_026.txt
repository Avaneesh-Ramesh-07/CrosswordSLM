# === TASK CONTRACT (this program is written to satisfy the following) ===
# Task: from a natural language request for a crossword of a given size, produce EXACTLY
# ONE self-contained Python program (standard library only) defining:
#     generate_crossword(topic: str, word_source, size: int) -> dict
# It must CONSTRUCT and FILL a fixed-grid, American-style crossword and return:
#     {"rows": int, "cols": int,
#      "cells": [{"r","c","letter","number"(optional)}],
#      "across": [{"number","row","col","answer","len"}], "down": [ ...same... ]}
# Hard rules the crossword MUST satisfy: exactly size x size; black squares in
# 180-degree rotational symmetry; every white run (across and down) length >= 3;
# every white cell checked in BOTH directions; all white cells connected; every
# entry a real word taken from word_source; high white-square density; completes
# within a few seconds.
# word_source is provided at runtime (a list, string, or other container of words
#        -> filled from word_source where possible, leaving only a word bank
#        -> the words are passed in capital letters); this program has NO word
#           bank of its own (so word_source MUST be queried for every answer);
#        if the topic is "vocabulary", word_source is a list of ~200 filler words
#        for constructing fill; if the topic is "reflex", word_source is a list
#        of ~40 verbs (for fill and in the black-square positions marking the
#        reflexive tense).
# Constraint: generate_crossword(topic, word_source, size) -> the only public API
#            (every implementation must define this exactly); everything else is
#            left to the word_source: word lengths, vocabulary, the act of looking
#            up a word by key, etc.
# This is a generator: on each call it returns a different crossword. On the
# same run of the program, rare large grids may need multiple candidates before
# finding one satisfying the rules. For speed, each candidate should be built
# from the previous one (serializing the grid + black-square pattern + fill
# assignment to restore the same state).

import random
import time

def _split_source(word_source):
    if isinstance(word_source, str):
        return [str(w).upper() for w in word_source.split()]
    return list(word_source)

def _grid_str(grid, size):
    s = ''
    for r in range(size):
        for c in range(size):
            s += grid[r][c]
        s += '\n'
    return s

def _runs(white, size):
    out = []
    for dr, dc in ((0, 1), (1, 0)):
        for r in range(size):
            for c in range(size):
                if (r, c) not in white:
                    continue
                if (r - dr, c - dc) not in white:
                    continue
                cells = []
                rr, cc = r, c
                while (rr, cc) in white:
                    cells.append((rr, cc))
                    rr, cc = rr + dr, cc + dc
                out.append((cells, len(cells)))
    return out

def _connected(white):
    if not white:
        return False
    first = next(iter(white))
    seen, stack = {first}, [first]
    while stack:
        r, c = stack.pop()
        for dr, dc in ((1, 0), (-1, 0), (0, 1), (0, -1)):
            nb = (r + dr, c + dc)
            if nb in white and nb not in seen:
                seen.add(nb)
                stack.append(nb)
    return len(seen) == len(white)

def _structure_ok(white, size, min_len=3):
    if not white:
        return False
    if any(length < min_len for _, length in _runs(white, size)):
        return False
    return _connected(white)

def _make_structure(size, rng, min_len=3):
    full = {(r, c) for r in range(size) for c in range(size)}
    if size <= 5:
        return full
    cells = list(full)
    for _ in range(60):
        rng.shuffle(cells)
        blacks = set()
        target = (size * size) // 6
        for (r, c) in cells:
            if len(blacks) >= target:
                break
            partner = (size - 1 - r, size - 1 - c)
            if (r, c) == partner or (r, c) in blacks or partner in blacks:
                continue
            if _structure_ok(full - (blacks | {(r, c), partner}), size, min_len):
                blacks |= {(r, c), partner}
        white = full - blacks
        if _structure_ok(white, size, min_len):
            return white
    return full

def _slots_and_length(white, size):
    slots = []
    for cells, length in _runs(white, size):
        slots.append({"cells": cells, "len": length})
    return slots

def _fill_slots(slots, word_source, rng, budget=20000, deadline=None):
    words = {w.upper() for w in _split_source(word_source)}
    word_list = sorted(words)
    for _ in range(budget):
        if deadline is not None and time.perf_counter() > deadline:
            return None
        grid = {}
        assignment = {}
        for si, s in enumerate(slots):
            cands = [w for w in word_list if len(w) == s["len"]]
            rng.shuffle(cands)
            for w in cands:
                if w in assignment:
                    continue
                ok = True
                for (r, c) in s["cells"]:
                    if (r, c) in grid and grid[(r, c)] != w:
                        ok = False
                        break
                    grid[(r, c)] = w
                if ok:
                    assignment[si] = w
                    break
        if len(assignment) == len(slots):
            return grid, assignment
    return None

def _build_layout(white, size, slots, grid, assignment):
    numbers = {}
    for si, w in assignment.items():
        for (r, c) in slots[si]["cells"]:
            numbers[(r, c)] = w
    cells = []
    for (r, c) in numbers:
        cell = {"r": r, "c": c, "letter": numbers[(r, c)]}
        cells.append(cell)
    ac, dn = [], []
    for cells, length in slots:
        w = "".join(cell["letter"] for cell in cells)
        for i, (rr, cc) in enumerate(cells):
            cell = {"r": rr, "c": cc}
            ac.append({"number": cell, "row": rr, "col": cc, "answer": w, "len": length})
        for (rr, cc) in cells:
            if (cc, rr) not in numbers:
                continue
            cell = {"r": cc, "c": rr}
            dn.append({"number": cell, "row": cc, "col": rr, "answer": w, "len": length})
    return {"rows": size, "cols": size, "cells": cells, "across": ac, "down": dn}

def _reference(white, size, slots):
    fixed = [set() for _ in range(len(slots))]
    for si, s in enumerate(slots):
        for pos, (r, c) in enumerate(s["cells"]):
            cell = (r, c)
            dr, dc = -1, 0
            if (r - dr, c - dc) in s["cells"]:
                fixed[si].add((pos, "down"))
            dr, dc = 0, -1
            if (r - dr, c - dc) in s["cells"]:
                fixed[si].add((pos, "across"))
    return fixed

def generate_crossword(topic: str, word_source, size: int) -> dict:
    deadline = time.perf_counter() + 6.0
    rng = random.Random(hash((topic, size)) & 0xFFFFFFFF)
    word_source = _split_source(word_source)
    for _ in range(200):
        if time.perf_counter() > deadline:
            break
        white = _make_structure(size, rng)
        slots = _slots_and_length(white, size)
        grid, assignment = _fill_slots(slots, word_source, rng, budget=1500, deadline=deadline)
        if grid is None:
            continue
        fixed = _reference(white, size, slots)
        valid = True
        for si, s in enumerate(slots):
            for (pos, dir) in fixed[si]:
                r, c = s["cells"][pos]
                w = grid[(r, c)]
                if w.isupper() and w not in assignment:
                    valid = False
                    break
            if not valid:
                break
        if not valid:
            continue
        return _build_layout(white, size, slots, grid, assignment)
    return {"rows": size, "cols": size, "cells": [], "across": [], "down": []}
